        function switchTab(mode) {
            const loginForm = document.getElementById('form-login');
            const newUserForm = document.getElementById('form-newuser');
            const loginTab = document.getElementById('tab-login');
            const newUserTab = document.getElementById('tab-newuser');

            if (mode === 'login') {
                loginForm.classList.remove('hidden');
                newUserForm.classList.add('hidden');
                
                loginTab.classList.add('bg-secondary-container', 'text-on-secondary-container', 'shadow-lg');
                loginTab.classList.remove('text-on-surface-variant');
                
                newUserTab.classList.remove('bg-secondary-container', 'text-on-secondary-container', 'shadow-lg');
                newUserTab.classList.add('text-on-surface-variant');
            } else {
                loginForm.classList.add('hidden');
                newUserForm.classList.remove('hidden');
                
                newUserTab.classList.add('bg-secondary-container', 'text-on-secondary-container', 'shadow-lg');
                newUserTab.classList.remove('text-on-surface-variant');
                
                loginTab.classList.remove('bg-secondary-container', 'text-on-secondary-container', 'shadow-lg');
                loginTab.classList.add('text-on-surface-variant');
            }
        }

        // Simple animation logic for the background atmosphere
        document.addEventListener('mousemove', (e) => {
            const x = e.clientX / window.innerWidth;
            const y = e.clientY / window.innerHeight;
            
            const atmosphere = document.querySelectorAll('.fixed.inset-0 .rounded-full');
            atmosphere[0].style.transform = `translate(${x * 50}px, ${y * 50}px)`;
            atmosphere[1].style.transform = `translate(${-x * 50}px, ${-y * 50}px)`;
        });
    </script>
